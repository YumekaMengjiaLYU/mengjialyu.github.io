library(tidyverse)
library(viridis)
library(p8105.datasets)
library(plotly)
data("rest_inspec")
rest_inspec = 
  rest_inspec %>%
  select(
    dba, boro, cuisine_description, inspection_date, grade) %>%
  filter(  
    grade != "NA",
    
    cuisine_description == "Chinese",
    format(as.Date(inspection_date, format = "%Y-%m-%d"), "%Y") == "2017"
  ) %>%
  distinct()
rest_inspec %>% 
  plot_ly(x = ~boro, y = ~grade, type = "scatter", mode = "markers",
          color = ~dba, alpha = 0.5)
