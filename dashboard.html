scatter_plot = ny_noaa %>%

  scale_color_viridis() +
  coord_cartesian() +
  theme_classic() +
  theme(axis.text.x  = element_text(angle = 90, hjust = 1)) +
  ggtitle("Relationship Between Minimum Temp & Maximum Temp in NY,1994")
